\newcommand{\einheit}[1]{\renewcommand{\nomentryend}{\hfill $\left[ #1 \right]$} } % Einheiten können angegeben werden 

% ++++ Damit Nomenklatur im Anhang als eigenes Chapter aufgeführt wird 
\setlength{\nomlabelwidth}{20mm}  % LabelBreite
\makeatletter   
\renewcommand{\thenomenclature}{%
\chapter{\nomname}
\list{}{
\labelwidth\nom@tempdim
\leftmargin\labelwidth
\advance\leftmargin\labelsep
\let\makelabel\nomlabel
}
}
\makeatother

	\ifthenelse{\equal{#1}{S}}{\item[\textbf{List of Symbols}]}{%
		\ifthenelse{\equal{#1}{A}}{\item[\textbf{List of Abbreviations}]} {
		  \ifthenelse{\equal{#1}{I}}{\item[\textbf{List of indices}]} {} } }	
		 }
